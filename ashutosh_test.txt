--22.	Start the executable section with the BEGIN keyword and include a SELECT statement to retrieve the maximum 
--department_id from the departments table.
SET SERVEROUTPUT ON
DECLARE
v_max NUMBER(8,2);
BEGIN
SELECT MAX(DEPARTMENT_ID)
INTO v_max
FROM TBLDEPARTMENTS;
DBMS_OUTPUT.PUT_LINE(v_max);
END;

--24.	Write PL/SQL blocks to show the scope and visibility of local and global identifiers.
SET SERVEROUTPUT ON
DECLARE
v_num VARCHAR2(20):='GLOBAL_VARIABLE';
BEGIN
DBMS_OUTPUT.PUT_LINE(v_num);
DECLARE
v_num1 VARCHAR2(20):='LOCAL_VARIABLE';
BEGIN
DBMS_OUTPUT.PUT_LINE(v_num1);
END;
END;

--25.	Write a PL/SQL block to adjust the salary of the employee whose ID 122.
SET SERVEROUTPUT ON
DECLARE
v_sal NUMBER(10):=100;
BEGIN
SELECT SALARY
INTO v_sal
FROM TBLEMPLOYEES
WHERE EMPLOYEE_ID=122;
DBMS_OUTPUT.PUT_LINE(v_sal);
END;
